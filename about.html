JavaScript Example 
// Simple API integration
const clearbg = require('clearbg-api');

// Initialize client
const client = new clearbg.Client({
  apiKey: 'your-api-key',
  baseUrl: 'https://api.clearbg.com/v1'
});

// Remove background
async function removeBackground(imagePath) {
  try {
    const result = await client.removeBackground({
      image: imagePath,
      format: 'png',
      size: '4k',
      enhance: true
    });
    
    console.log('Processing complete:', result);
    return result.url;
  } catch (error) {
    console.error('Processing failed:', error);
  }
}
